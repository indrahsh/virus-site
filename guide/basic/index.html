Basic

Comments

A comment.

// A single line comment.

Values

In Virus there are 8 types of value: Null, Boolean, Number, Type, Object, String, Array, and HashTable.

Null

null

Boolean

false and true

Number

A Number.

10
// 10

-5
// -5

0.5
// 0.5

1 000 000
// 1000000

String

A String.

"Hello, world!"
// Hello, world!

A multiline String.

string = (
  "Hello, " +
  "world!"
)
// Hello, world!

Array

An Array is a collection of values identified by integer index. Negative indices work backwards from the end.

array Array<String> = <String>["Foo", "Bar"]
// ["Foo", "Bar"]

array Array<String> = <String>[]

array[0] = "Foo"
array[1] = "Bar"
// ["Foo", "Bar"]

HashTable

A HashTable is a collection of key-value pairs, the key can be a Boolean, Number, String, Type, or an Object. Array and HashTable can't be used as keys.

hashTable HashMap<String, Number> = <String, Number>{"Foo": 1, "Bar": 2}
// {"Foo": 1, "Bar": 2}

hashTable HashMap<String, Number> = <String, Number>{}

hashTable["Foo"] = 1
hashTable["Bar"] = 2
// {"Foo": 1, "Bar": 2}

Object

Every value is an Object.

2.pow(2).pow(2)
// 16

"Foo Bar".Split(" ")
// ["Foo", "Bar"]

[1, 2].Length
// 2

{"Foo": 1, "Bar": 2}.Length
// 2

Control Flow

Conditional

if.

if true
    ;

else.

if false
    ;
else
    ;

else if.

if false
    ;
else if false
    ;
else
    ;

Loops

while.

count = 0
while count < 5
    count += 1
    System.Print(count)
// 1, 2, 3, 4, 5

for each.

array Array<Number> = <Number>[1, 2, 3, 4, 5]
for each value in array
    System.Print(value)
// 1, 2, 3, 4, 5

for each key, value in array
    System.Print("%: %".Format(key, value))
// 0: 1, 1: 2, 2: 3, 3: 4, 4: 5

Type

Defining a Type and its constructor method.

object Vehicle
    ()
        ;

Instantiating a Type.

vehicle = Vehicle()
// Vehicle

Defining a property.

object Vehicle
    _speed Number = 0
    ()
        ;

vehicle = Vehicle()
vehicle.speed
// Error: Instance of Vehicle has no method speed.

Defining a setter and a getter.

object Vehicle
    _speed Number = 0
    ()
        ;
    speed=(value Number)
        _speed = speed
    speed
        return _speed

vehicle = Vehicle()
vehicle.speed
// 0
vehicle.speed = vehicle.speed + 5
// 5

Defining a method.

object Vehicle
    _x Number = 0
    _y Number = 0
    ()
        ;
    x Number
        return _x
    y Number
        return _y
    move(x Number, y Number)
        _x += x
        _y += y

vehicle = Vehicle()
vehicle.move(5, 5)
vehicle.x
// 5
vehicle.y
// 5

Extending a Type.

object Vehicle
    _x Number = 0
    _y Number = 0
    ()
        ;
    x Number
        return _x
    y Number
        return _y
    move(x Number, y Number)
        _x += x
        _y += y

object Carriage Vehicle
    ()
        super()

carriage = Carriage()
carriage.move(5, 5)
carriage.x
// 5
carriage.y
// 5

The super keyword.

object Vehicle
    _x Number = 0
    _y Number = 0
    ()
        ;
    x Number
        return _x
    y Number
        return _y
    move(x Number, y Number)
        _x += x
        _y += y

object Ship Vehicle
    move(x Number, y Number)
        return super(x, y)

ship = Ship()
ship.move(5, 5)
ship.x
// 5
ship.y
// 5

API

Store user data using Virus_State_SetUserData, Virus_State_GetUserData, Virus_State_SetSlotObjectUserData, and Virus_State_GetSlotObjectUserData.

Bind native methods with Virus_State_BindUserStaticNativeFunction and Virus_State_BindUserNativeFunction.

Creating and reading a Boolean.

Virus_State_SetSlotFalse(state, 0);
bool result = Virus_State_GetSlotBoolean(state, 0);

Creating and reading a Number.

Virus_State_SetSlotNumber(state, 0, 1.0);
double result = Virus_State_GetSlotDouble(state, 0, &result);

Creating and reading a String.

Virus_State_SetSlotNewString(state, 0, "Hello, world!");
const char *result = Virus_State_GetSlotString(state, 0);

Creating an Array.

Virus_State_SetSlotNewArray(state, 0);

Setting a value in an Array.

Virus_State_SetSlotNewString(state, 1, "Foo")
Virus_State_SetSlotArrayValue(state, 0, 0, 1);

Getting a value in an Array.

Virus_State_GetSlotArrayValue(state, 0, 0, 1);
const char *result = Virus_State_GetSlotDouble(state, 1);

Creating a HashTable.

Virus_State_SetSlotNewHashTable(state, 0);

Setting a value in a HashTable.

Virus_State_SetSlotNewString(state, 1, "Foo");
Virus_State_SetSlotNumber(state, 2, 1);
Virus_State_SetSlotHashTableValue(state, 0, 1, 2);

Getting a value in a HashTable.

Virus_State_GetSlotHashTableValue(state, 0, 1, 2);
double result = Virus_State_GetSlotDouble(state, 2);

Getting a Type.

Virus_State_SetSlotModule(state, 0, "Math");
Virus_State_SetSlotType(state, 1, 0, "Vector3");

Instantiating an Object.

Virus_State_SetSlotModule(state, 0, "Math");
Virus_State_SetSlotType(state, 1, 0, "Vector3");
Virus_State_SetSlotNumber(state, 2, 0.0);
Virus_State_SetSlotNumber(state, 3, 0.0);
Virus_State_SetSlotNumber(state, 4, 0.0);
Virus_State_SetSlotNewObject(state, 5, 1, "Vector3(Number, Number, Number)");
Virus_ValueHandle *vectorObjectHandle = Virus_State_CreateSlotValueHandle(state, 5);

Calling a static method.

Virus_State_SetSlotNumber(state, 1, 1.0);
Virus_State_SetSlotNumber(state, 2, 1.0);
Virus_State_SetSlotCallMethod(state, 1, "+");
double result = Virus_State_GetSlotDouble(state, 1);
